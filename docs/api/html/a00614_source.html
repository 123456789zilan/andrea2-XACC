158  template<class BoostVertex>
159  void operator()(std::ostream& out, const BoostVertex& v) const {
160  auto node = vertex(v, graph);
161  std::stringstream ss;
162  ss << graph[node].properties;
163 
164  // Now we have a string of comma separated parameters
165  std::string result;
166  auto vertexString = ss.str();
167  boost::trim(vertexString);
168  std::vector<std::string> splitVec;
169  boost::split(splitVec, vertexString, boost::is_any_of(","));
170 
171  int counter = 0;
172  for (std::size_t i = 0; i < splitVec.size(); i++) {
173  if (!graph[node].propertyNames[counter].empty()) {
174  auto s = splitVec[i];
175  if (i == 0) {
176  s.insert(8, graph[node].propertyNames[counter] + "=");
177  } else {
178  s.insert(0, graph[node].propertyNames[counter] + "=");
179  }
180  counter++;
181  result += s + ",";
182  }
183  }
184 
185  result = result.substr(0,result.size() - 1);
186  if (result.substr(result.size() - 2, result.size()) != "\"]") {
187  result += "\"]";
188  }
189  out << " " << result;
190  }
191  };

375  int diameter() {
376  // Set some aliases we'll need
377  using DistanceProperty = boost::exterior_vertex_property<adj_list, int>;
378  using DistanceMatrix = typename DistanceProperty::matrix_type;
379  using DistanceMatrixMap = typename DistanceProperty::matrix_map_type;
380  using EccentricityProperty = boost::exterior_vertex_property<adj_list, int>;
381  using EccentricityContainer = typename EccentricityProperty::container_type;
382  using EccentricityMap = typename EccentricityProperty::map_type;
383 
384  // Construct the distance mapping
385  DistanceMatrix distances(order());
386  DistanceMatrixMap dm(distances, *_graph.get());
387  constant_property_map<edge_type, int> wm(1);
388 
389  // Compute the shortest paths
390  floyd_warshall_all_pairs_shortest_paths(*_graph.get(), dm, weight_map(wm));
391 
392  // Now get diameter information
393  EccentricityContainer eccs(order());
394  EccentricityMap em(eccs, *_graph.get());
395 
396  // Return the diameter
397  return all_eccentricities(*_graph.get(), dm, em).second;
398  }
