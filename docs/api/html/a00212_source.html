197 template<typename Signature>
198 auto mangled_storage_impl::get_constructor() const -> ctor_sym
199 {
200  using namespace parser;
201 
202  using func_type = Signature*;
203 
204  std::string ctor_name; // = class_name + "::" + name;
205  std::string unscoped_cname; //the unscoped class-name
206  {
207  auto class_name = get_return_type(dummy<Signature>());
208  auto pos = class_name.rfind("::");
209  if (pos == std::string::npos)
210  {
211  ctor_name = class_name+ "::" +class_name ;
212  unscoped_cname = class_name;
213  }
214  else
215  {
216  unscoped_cname = class_name.substr(pos+2) ;
217  ctor_name = class_name+ "::" + unscoped_cname;
218  }
219  }
220 
221  auto matcher =
222  ctor_name + '(' + parser::arg_list(*this, func_type()) + ')';
223 
224 
225  std::vector<entry> findings;
226  std::copy_if(storage_.begin(), storage_.end(),
227  std::back_inserter(findings), [&](const entry& e) {return e.demangled == matcher;});
228 
229  ctor_sym ct;
230 
231  for (auto & e : findings)
232  {
233 
234  if (e.mangled.find(unscoped_cname +"C1E") != std::string::npos)
235  ct.C1 = e.mangled;
236  else if (e.mangled.find(unscoped_cname +"C2E") != std::string::npos)
237  ct.C2 = e.mangled;
238  else if (e.mangled.find(unscoped_cname +"C3E") != std::string::npos)
239  ct.C3 = e.mangled;
240  }
241  return ct;
242 }

246 {
247  std::string dtor_name; // = class_name + "::" + name;
248  std::string unscoped_cname; //the unscoped class-name
249  {

251  auto pos = class_name.rfind("::");
252  if (pos == std::string::npos)
253  {
254  dtor_name = class_name+ "::~" + class_name + "()";
255  unscoped_cname = class_name;
256  }
257  else
258  {
259  unscoped_cname = class_name.substr(pos+2) ;
260  dtor_name = class_name+ "::~" + unscoped_cname + "()";
261  }
262  }
263 
264  auto d0 = unscoped_cname + "D0Ev";
265  auto d1 = unscoped_cname + "D1Ev";
266  auto d2 = unscoped_cname + "D2Ev";

268  dtor_sym dt;
269  //this is so simple, i don#t need a predicate
270  for (auto & s : storage_)

272  //alright, name fits
