66  std::string write() {
67  // This is a Function, start it as an Object
68  writer->StartObject();
69  writer->String("function");
70  writer->String(function->getName());
71 
72  // All functions have instructions, start
73  // that array here.
74  writer->String("instructions");
75  writer->StartArray();
76 
77  topLevelInstructionIterator = std::make_shared<xacc::InstructionIterator>(function);
78  while (topLevelInstructionIterator->hasNext()) {
79  // Get the next node in the tree
80  auto nextInst = topLevelInstructionIterator->next();
81  nextInst->accept(this);
82  }
83 
84  // End Instructions
85  writer->EndArray();
86 
87  // End Function
88  writer->EndObject();
89  return buffer->GetString();
90  }
