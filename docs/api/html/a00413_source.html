113 run(Vertex s)
114 {
115  double inf = (std::numeric_limits<double>::max)();
116 
117  auto local = get(vertex_index, g);
118  // None of the vertices are stored in the bucket.
119  position_in_bucket.clear();
120  position_in_bucket.resize(num_vertices(g), dummy_list.end());
121 
122  // None of the vertices have been deleted
123  vertex_was_deleted.clear();
124  vertex_was_deleted.resize(num_vertices(g), false);
125 
126  // No path from s to any other vertex, yet
127  BGL_FORALL_VERTICES_T(v, g, Graph) {
128  put(distance, v, inf);
129  }
130 
131  // Put "s" into its bucket (bucket 0)
132  relax(s, s, 0);
133 
134  BucketIndex max_bucket = (std::numeric_limits<BucketIndex>::max)();
135  BucketIndex current_bucket = 0;
136  do {
137  // Synchronize with all of the other processes.
138  synchronize();
139 
140  // Find the next bucket that has something in it.
141  while (current_bucket < buckets.size()
142  && (!buckets[current_bucket] || buckets[current_bucket]->empty()))
143  ++current_bucket;
144  if (current_bucket >= buckets.size())
145  current_bucket = max_bucket;
146 
147  // Find the smallest bucket (over all processes) that has vertices
148  // that need to be processed.
149 // using boost::parallel::all_reduce;
150 // using boost::parallel::minimum;
151 // current_bucket = all_reduce(pg, current_bucket, minimum<BucketIndex>());
152 
153  if (current_bucket == max_bucket)
154  // There are no non-empty buckets in any process; exit.
155  break;
156 
157  // Contains the set of vertices that have been deleted in the
158  // relaxation of "light" edges. Note that we keep track of which
159  // vertices were deleted with the property map
160  // "vertex_was_deleted".
161  std::vector<Vertex> deleted_vertices;
162 
163  // Repeatedly relax light edges
164  bool nonempty_bucket;
165  do {
166  // Someone has work to do in this bucket.
167 
168  if (current_bucket < buckets.size() && buckets[current_bucket]) {
169  Bucket& bucket = *buckets[current_bucket];
170  // For each element in the bucket
171  while (!bucket.empty()) {
172  Vertex u = bucket.front();
173 
174  // Remove u from the front of the bucket
175  bucket.pop_front();
176 
177  // Insert u into the set of deleted vertices, if it hasn't
178  // been done already.
179  if (!vertex_was_deleted[get(local, u)]) {
180  vertex_was_deleted[get(local, u)] = true;
181  deleted_vertices.push_back(u);
182  }
183 
184  // Relax each light edge.
185  double u_dist = get(distance, u);
186  BGL_FORALL_OUTEDGES_T(u, e, g, Graph) {
187  Vertex tgt = target(e, g);
188  if (get(bias, tgt) <= delta) // light edge
189  relax(u, tgt, u_dist + get(bias, tgt));
190  }
191  }
192  }
193 
194  // Synchronize with all of the other processes.
195  synchronize();
196 
197  // Is the bucket empty now?
198  nonempty_bucket = (current_bucket < buckets.size()
199  && buckets[current_bucket]
200  && !buckets[current_bucket]->empty());
201  } while (nonempty_bucket);
202 
203  // Relax heavy edges for each of the vertices that we previously
204  // deleted.
205  for (typename std::vector<Vertex>::iterator iter = deleted_vertices.begin();
206  iter != deleted_vertices.end(); ++iter) {
207  // Relax each heavy edge.
208  Vertex u = *iter;
209  double u_dist = get(distance, u);
210  BGL_FORALL_OUTEDGES_T(u, e, g, Graph) {
211  Vertex tgt = target(e, g);
212  if (get(bias, tgt) > delta) // heavy edge
213  relax(u, tgt, u_dist + get(bias, tgt));
214  }
215  }
216 
217  // Go to the next bucket: the current bucket must already be empty.
218  ++current_bucket;
219  } while (true);
220 
221  // Delete all of the buckets.
222  for (typename std::vector<Bucket*>::iterator iter = buckets.begin();
223  iter != buckets.end(); ++iter) {
224  if (*iter) {
225  delete *iter;
226  *iter = 0;
227  }
228  }

233 void
234 VertexBiasDeltaStepping<Graph, PredecessorMap, DistanceMap, BiasMap>::
235 relax(Vertex u, Vertex v, double x)
236 {
237  auto local = get(vertex_index, g);
238  if (x < get(distance, v)) {
239  // We're relaxing the edge to vertex v.
240  // Compute the new bucket index for v
241  BucketIndex new_index = static_cast<BucketIndex>(x / delta);
242 
243  // Make sure there is enough room in the buckets data structure.
244  if (new_index >= buckets.size()) buckets.resize(new_index + 1, 0);
245 
246  // Make sure that we have allocated the bucket itself.
247  if (!buckets[new_index]) buckets[new_index] = new Bucket;
248 
249  if (get(distance, v) != (std::numeric_limits<double>::max)()
250  && !vertex_was_deleted[get(local, v)]) {
251 
252  // We're moving v from an old bucket into a new one. Compute
253  // the old index, then splice it in.
254  BucketIndex old_index
255  = static_cast<BucketIndex>(get(distance, v) / delta);
256  buckets[new_index]->splice(buckets[new_index]->end(),
257  *buckets[old_index],
258  position_in_bucket[get(local, v)]);
259  } else {
260  // We're inserting v into a bucket for the first time. Put it
261  // at the end.
262  buckets[new_index]->push_back(v);
263  }
264 
265  // v is now at the last position in the new bucket
266  position_in_bucket[get(local, v)] = buckets[new_index]->end();
267  --position_in_bucket[get(local, v)];
268 
269  // Update predecessor and tentative distance information
270  put(predecessor, v, u);
271  put(distance, v, x);
272  }
273 }
