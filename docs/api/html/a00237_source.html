101  void applyUnitary(fire::Tensor<2, fire::EigenProvider, std::complex<double>>& U) {
102  assert(
103  U.dimension(0) == bufferState.dimension(0)

105  std::array<IndexPair, 1> contractionIndices;
106  contractionIndices[0] = std::make_pair(1, 0);
107  bufferState = U.contract(bufferState, contractionIndices);

113  void normalize() {
114  double sum = 0.0;
115  for (int i = 0; i < bufferState.dimension(0); i++)
116  sum += std::real(bufferState(i) * bufferState(i));
117  for (int i = 0; i < bufferState.dimension(0); i++)
118  bufferState(i) = bufferState(i) / std::sqrt(sum);
119  }
