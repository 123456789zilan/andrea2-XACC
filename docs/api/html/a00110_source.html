101  ChimeraGraph(int gridSize) : DWGraph(8 * gridSize * gridSize) {
102  // We are going to assign a tuple of ints to each vertex
103  std::map<std::tuple<int, int, int, int>, int> vertexTuples;
104  int i, j, k, l, v = 0;
105 
106  // Loop through the list of vertices and give
107  // them the correct 4-tuple: (i,j) is unit cell in
108  // the chimera grid, (k,l) is vertex in unit cell, k = 0-3,
109  // l = 0,1 (left, right of bipartite graph)
110  for (i = 0; i < gridSize; i++) {
111  for (j = 0; j < gridSize; j++) {
112  for (l = 0; l < 2; l++) {
113  for (k = 0; k < 4; k++) {
114  vertexTuples.insert(
115  std::make_pair(std::make_tuple(i, j, k, l), v));
116  v++;
117  }
118  }
119  }
120  }
121 
122  // Setup edges within each cell
123  for (i = 0; i < gridSize; i++) {
124  for (j = 0; j < gridSize; j++) {
125  for (k = 0; k < 4; k++) {
126  for (int k2 = 0; k2 < 4; k2++) {
127  int v1 = vertexTuples[std::make_tuple(i, j, k, 0)];
128  int v2 = vertexTuples[std::make_tuple(i, j, k2, 1)];
129  addEdge(v1, v2);
130  }
131 
132  }
133  }
134  }
135 
136  // Setup edges between cells
137  for (i = 0; i < gridSize - 1; i++) {
138  for (j = 0; j < gridSize; j++) {
139  for (k = 0; k < 4; k++) {
140  int v1 = vertexTuples[std::make_tuple(i, j, k, 0)];
141  int v2 = vertexTuples[std::make_tuple(i + 1, j, k, 0)];
142  addEdge(v1, v2);
143 
144  }
145  }
146  }
147 
148  for (i = 0; i < gridSize; i++) {
149  for (j = 0; j < gridSize - 1; j++) {
150  for (k = 0; k < 4; k++) {
151  int v1 = vertexTuples[std::make_tuple(i, j, k, 1)];
152  int v2 = vertexTuples[std::make_tuple(i, j + 1, k, 1)];
153  addEdge(v1, v2);

155  }
