147  virtual void initialize() {
148  auto options = RuntimeOptions::instance();
149  searchAPIKey(apiKey, url);
150  auto tempURL = url;
151  boost::replace_all(tempURL, "https://", "");
152  boost::replace_all(tempURL, "/sapi", "");
153 
154  // Set up the extra HTTP headers we are going to need
155  headers.insert(std::make_pair("X-Auth-Token", apiKey));
156  headers.insert(std::make_pair("Content-type", "application/x-www-form-urlencoded"));
157  headers.insert(std::make_pair("Accept", "*/*"));
158 
159  // Get the Remote URL Solver data...
160  auto getSolverClient = fire::util::AsioNetworkingTool<SimpleWeb::HTTPS>(tempURL, false);
161  auto r = getSolverClient.get("/sapi/solvers/remote", headers);
162 
163  std::stringstream ss;
164  ss << r.content.rdbuf();
165  auto message = ss.str();
166 
167  Document document;
168  document.Parse(message.c_str());
169 
170  if (document.IsArray()) {
171  for (auto i = 0; i < document.Size(); i++) {
172  DWSolver solver;
173  solver.name = document[i]["id"].GetString();
174  boost::trim(solver.name);
175  solver.description = document[i]["description"].GetString();
176  if (document[i]["properties"].FindMember("j_range") != document[i]["properties"].MemberEnd()) {
177  solver.jRangeMin = document[i]["properties"]["j_range"][0].GetDouble();
178  solver.jRangeMax = document[i]["properties"]["j_range"][1].GetDouble();
179  solver.hRangeMin = document[i]["properties"]["h_range"][0].GetDouble();
180  solver.hRangeMax = document[i]["properties"]["h_range"][1].GetDouble();
181 
182  }
183  solver.nQubits = document[i]["properties"]["num_qubits"].GetInt();
184  std::cout << "INSERTING: " << solver.name << "\n";
185  availableSolvers.insert(std::make_pair(solver.name, solver));
186  }
187  }
188  }

190  virtual std::shared_ptr<AcceleratorBuffer> createBuffer(
191  const std::string& varId) {
192  auto options = RuntimeOptions::instance();
193  std::string solverName = "DW_2000Q_VFYC";
194  if (options->exists("dwave-solver")) {
195  solverName = (*options)["dwave-solver"];
196  }
197  if (!availableSolvers.count(solverName)) {
198  XACCError(solverName + " is not available for creating a buffer.");
199  }
200  auto solver = availableSolvers[solverName];
201  auto buffer = std::make_shared<AcceleratorBuffer>(varId, solver.nQubits);
202  storeBuffer(varId, buffer);
203  return buffer;
204  }
